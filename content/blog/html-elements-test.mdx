```javascript
function calculateTotal(items) {
  return items
    .filter(item => item.inStock)
    .reduce((total, item) => {
      return total + (item.price * item.quantity);
    }, 0);
}

const cart = [
  { name: 'Widget', price: 9.99, quantity: 2, inStock: true },
  { name: 'Gadget', price: 19.99, quantity: 1, inStock: true },
  { name: 'Doohickey', price: 29.99, quantity: 3, inStock: false }
];

console.log(`Total: $${calculateTotal(cart).toFixed(2)}`);
```
